2 — JS: TM conditional visibility
var _tmEnabled = false;
function _applyTmVisibility() {
  var wrap = document.getElementById('d-svc-tm-wrap');
  if (!wrap) return;
  wrap.style.display = _tmEnabled ? '' : 'none';
  if (!_tmEnabled) {
    var tmChk = document.getElementById('d-svc-tm');
    if (tmChk) tmChk.checked = false;
  }
}
// Called on Firebase login — reads company feature flags
window._fbOnLogin = function(user) {
  // ... loadDrivers, loadVehicles, loadDispatchers ...
  window.adminRead('companySettings/' + COMPANY_ID + '/features').then(function(f) {
    _tmEnabled = !!(f && (f.tmEnabled || f.totalMobility));  // supports both flag names
  }).catch(function() { _tmEnabled = false; });
};

3 — JS: Reading saved services when opening Edit Driver
// Supports both old single string and new comma-separated format
(function() {
  var saved = d.foodDelivery || '';
  var vals  = saved ? saved.split(',') : [];
  // Legacy value mapping (old food delivery enum → new service keys)
  var legacyMap = {
    'all':      'taxi',
    'ubereats': 'food',
    'doordash': 'food',
    'menulog':  'food',
    'custom':   'food',
    'none':     ''
  };
  vals = vals.map(function(v) {
    return legacyMap.hasOwnProperty(v) ? legacyMap[v] : v;
  }).filter(Boolean);
  // Set checkboxes from old foodDelivery string
  ['taxi','food','freight','courier','airport','corporate',
   'school','disability','medical','event','tm'].forEach(function(k) {
    var el = document.getElementById('d-svc-' + k);
    if (el) el.checked = vals.indexOf(k) !== -1;
  });
  // Overlay with allowedServices boolean map if present (written by newer saves — takes priority)
  var as = d.allowedServices;
  if (as && typeof as === 'object' && !Array.isArray(as)) {
    ['taxi','food','freight','tm'].forEach(function(k) {
      var el = document.getElementById('d-svc-' + k);
      if (el && as.hasOwnProperty(k)) el.checked = !!as[k];
    });
  }
})();
